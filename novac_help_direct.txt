Nova Compiler v1.0.0

USAGE:
  nova <file>                    Parse and check file
  nova <file> -o <output>        Compile to executable
  nova build <file>              Build executable
  nova run <file>                JIT run or build+run
  nova check <file>              Type check only

OUTPUT CONTROL:
  -o <file>                      Output file
  --emit-ast                     Emit AST
  --emit-ir                      Emit LLVM IR
  --emit-obj                     Emit object file
  --emit-asm                     Emit assembly
  --dump-tokens                  Dump lexer tokens

EXECUTION:
  --jit                          Use JIT backend
  --run                          Run after compile

OPTIMIZATION:
  -O0, -O1, -O2, -O3            Optimization level

OPTIONS:
  --verbose, -v                  Verbose output
  --version                      Show version
  --help, -h                     Show this help
  --no-borrow-check              Disable borrow checker
  --deterministic                Deterministic build
